Deployment & Distribution

23.1 Production Build

# Build optimized bundle
bun scripts/build-bundle.ts --minify

# Package for npm
bun scripts/package-npm.ts

# Output:
# dist/npm/
# ├── cli.mjs          # Single-file bundled CLI
# ├── cli.mjs.map      # Source map
# ├── package.json      # npm metadata
# └── README.md

23.2 Docker Deployment

# Multi-stage build
FROM oven/bun:1 AS builder
# Install build tools, deps, build bundle

FROM oven/bun:1 AS runtime
# Copy only: node_modules (native), dist/, server/
# Create non-root user
# Health check: curl /health
# Entrypoint: validate config, setup data dir, start server

23.3 Environment Configuration

# Required
ANTHROPIC_API_KEY=sk-ant-...

# Optional: Provider override
CLAUDE_CODE_USE_BEDROCK=true
CLAUDE_CODE_USE_VERTEX=true
CLAUDE_CODE_USE_FOUNDRY=true

# Optional: Behavior
CLAUDE_CODE_SIMPLE=true           # Minimal mode (Bash, Read, Edit only)
CLAUDE_CODE_COORDINATOR_MODE=true # Multi-agent coordinator
CLAUDE_CODE_PROACTIVE=true        # Autonomous mode
