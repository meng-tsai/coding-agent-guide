Part 7 · Chapter 37

Practical Guide: Building Your Own Coding Agent

This section distills the lessons from Claude Code into actionable steps for building your own agent.

37.1 Minimum Viable Agent (Day 1)

// 1. System prompt
const systemPrompt = `
You are a coding assistant. You help users write and modify code.
You have access to tools for reading and writing files.

Rules:
- Read files before modifying them
- Make minimal changes to achieve the goal
- Don't add features beyond what was asked
`

// 2. Tool definitions (start with 3)
const tools = [
  {
    name: 'read_file',
    description: 'Read contents of a file',
    input_schema: { type: 'object', properties: { path: { type: 'string' } } },
    execute: async ({ path }) => readFileSync(path, 'utf-8')
  },
  {
    name: 'write_file',
    description: 'Write contents to a file',
    input_schema: { type: 'object', properties: { path: { type: 'string' }, content: { type: 'string' } } },
    execute: async ({ path, content }) => writeFileSync(path, content)
  },
  {
    name: 'run_command',
    description: 'Run a shell command',
    input_schema: { type: 'object', properties: { command: { type: 'string' } } },
    execute: async ({ command }) => execSync(command).toString()
  }
]

// 3. The loop
async function agentLoop(userMessage) {
  const messages = [{ role: 'user', content: userMessage }]

  while (true) {
    const response = await anthropic.messages.create({
      model: 'claude-sonnet-4-20250514',
      system: systemPrompt,
      tools: tools.map(t => ({ name: t.name, description: t.description, input_schema: t.input_schema })),
      messages
    })

    // Display text
    for (const block of response.content) {
      if (block.type === 'text') console.log(block.text)
    }

    // Execute tools
    const toolCalls = response.content.filter(b => b.type === 'tool_use')
    if (toolCalls.length === 0) break  // No more tool calls → done

    messages.push({ role: 'assistant', content: response.content })

    for (const call of toolCalls) {
      const tool = tools.find(t => t.name === call.name)
      try {
        const result = await tool.execute(call.input)
        messages.push({ role: 'user', content: [{ type: 'tool_result', tool_use_id: call.id, content: result }] })
      } catch (error) {
        messages.push({ role: 'user', content: [{ type: 'tool_result', tool_use_id: call.id, content: `Error: ${error.message}`, is_error: true }] })
      }
    }
  }
}

37.2 Add Safety (Week 1)

// Add permission checks before tool execution
async function checkPermission(tool, input) {
  // Read-only tools: always allow
  if (['read_file', 'glob', 'grep'].includes(tool.name)) return true

  // Destructive tools: ask user
  const answer = await prompt(`Allow ${tool.name}(${JSON.stringify(input)})? [y/n] `)
  return answer === 'y'
}

// Add tool result size limits
function truncateResult(result, maxChars = 30000) {
  if (result.length <= maxChars) return result
  return result.slice(0, maxChars) + '\n\n[... truncated ...]'
}

37.3 Add Context Management (Week 2)

// File state cache
const fileCache = new Map()

// Compact when approaching context limit
async function maybeCompact(messages) {
  const estimatedTokens = messages.reduce((sum, m) => sum + roughEstimate(m), 0)

  if (estimatedTokens > COMPACT_THRESHOLD) {
    const summary = await summarize(messages.slice(0, -10))
    return [{ role: 'user', content: summary }, ...messages.slice(-10)]
  }
  return messages
}

// Inject dynamic context
function buildSystemPrompt() {
  const gitStatus = execSync('git status --short').toString()
  const branch = execSync('git branch --show-current').toString().trim()

  return `${basePrompt}

# Environment
- Working directory: ${process.cwd()}
- Git branch: ${branch}
- Git status: ${gitStatus}
- Date: ${new Date().toISOString().split('T')[0]}
`
}

37.4 Add Multi-Agent Support (Month 1)

// Agent spawning with isolation
async function spawnAgent(prompt, tools, options = {}) {
  const agentContext = {
    fileCache: new Map(parentContext.fileCache),  // Clone, not share
    abortController: new AbortController(),
    maxTurns: options.maxTurns || 30,
  }

  return agentLoop(prompt, { tools, context: agentContext })
}

// Tool restriction for sub-agents
const readOnlyTools = tools.filter(t => t.isReadOnly)
const explorationAgent = spawnAgent('Explore the auth module', readOnlyTools)

37.5 Maturity Roadmap

Week 1:  Minimum viable agent (3 tools, basic loop)
Week 2:  Permission system (allow/deny rules)
Week 3:  File state cache + result truncation
Week 4:  Dynamic context (git, cwd, date)
Month 2: Context compaction (auto-summarize)
Month 2: Tool concurrency (parallel read-only tools)
Month 3: Sub-agent spawning with isolation
Month 3: Hook system (pre/post tool use)
